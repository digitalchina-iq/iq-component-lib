angular选择器

:host

/deep/

选择器应用样式到本身及子组件

/deep/ 简单示例，改变开关滑块颜色

 
            
              <iq-switcher [(ngModel)]="fromSwitcher"></iq-switcher>
            
          
            
              fromSwitcher: boolean;
            
          
            
              /deep/ .m-switch .switch-ball{{'{'}}
                background: red;
              {{'}'}}
