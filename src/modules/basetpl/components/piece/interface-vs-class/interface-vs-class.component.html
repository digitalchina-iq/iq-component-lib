interface 接口 和 class 类的区别，简单示例

  • 接口可以理解成一种特殊的类，由常量和抽象方法组成的特殊类。
  • 接口不能实例化
  • 接口只声明成员方法，不做实现，方法没有方法体{{'{'}}{{'}'}};
  • 继承接口的类一定要实现接口中定义的方法；
  • 类可以实例化，可以定义自己的字段，属性，方法等等；
  • 类可以继承多个接口，但只能继承一个类！
  • 类声明并实现方法。
 
            
            
          
            
interface Person {{'{'}}
  name: string;
  speak(): void;
{{'}'}}

class Student {{'{'}}
  name: string;
  gender: boolean;
  speak() {{'{'}}
    console.log('Hello, I\'m a student');
  {{'}'}}
{{'}'}}

class Teacher {{'{'}}
  name: string;
  age: number;
  speak() {{'{'}}
    console.log('Hello, I\'m a teacher');
  {{'}'}}
{{'}'}}

function sayHello(one: Person) {{'{'}}
  one.speak()
{{'}'}}

var s = new Student(),
    t = new Teacher();

sayHello(s);
sayHello(t);
